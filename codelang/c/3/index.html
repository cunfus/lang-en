线程

#include <stdio.h>
#include <pthread.h>
#include <unistd.h>
#include <signal.h>
#include <stdbool.h>

volatile sig_atomic_t stop = false;
volatile bool running = true;

void* worker(void *)
{
    while (running) {
        printf("thread running...\n");
        sleep(1);
    }

    printf("thread exit...\n");
    return NULL;
}

void signal_handle(int sig)
{
    stop = 1;
}

int main()
{
    pthread_t t;
    signal(SIGINT, signal_handle);

    pthread_create(&t, NULL, worker, NULL);

    while (!stop) {
        printf("Main Loop...\n");
        sleep(1);
    }

    running = false;

    pthread_join(t, NULL);

    return 0;
}
  • volatile 多用在多线程和信号的变量中，防止编译器优化，读到的是缓存值

怎么能确保信号被主线程捕获，而不是被子线程捕获？pthread_sigmask 在子线程中屏蔽特定信号。

    signal(SIGINT, signal_handle);

    sigset_t set;
    sigemptyset(&set);
    sigaddset(&set, SIGINT);
    pthread_sigmask(SIG_BLOCK, &set, NULL);

    pthread_create(&t, NULL, worker, NULL);

互斥锁

#include <stdio.h>
#include <pthread.h>
#include <unistd.h>
#include <signal.h>
#include <stdbool.h>

volatile sig_atomic_t stop = false;
volatile bool running = true;

pthread_mutex_t mutex;
int global_var = 0;

void* producer(void *)
{
    while (running) {
        pthread_mutex_lock(&mutex);
        global_var += 1;
        printf("[producer]: +1 -> %d\n", global_var);
        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    printf("producer exit...\n");
    return NULL;
}

void* consumer(void *)
{
    while (running) {
        pthread_mutex_lock(&mutex);
        global_var -= 1;
        printf("[consumer]: -1 -> %d\n", global_var);
        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    printf("consumer exit...\n");
    return NULL;
}

void signal_handle(int sig)
{
    stop = 1;
}

int main()
{
    pthread_t t1, t2;
    pthread_mutex_init(&mutex, NULL);
    signal(SIGINT, signal_handle);

    pthread_create(&t1, NULL, producer, NULL);
    pthread_create(&t2, NULL, consumer, NULL);

    while (!stop) {
        sleep(1);
    }

    running = false;

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    pthread_mutex_destroy(&mutex);
    return 0;
}

加锁是没问题了，在往上，还有 CPU 调度，哪个线程轮到的次数多还不一定。 如果实际上只需要一个线程写，其他线程读呢？怎么让读的人能一起看呢？互斥肯定不行，上读写锁吧。

读写锁

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

#define NUM_READERS 5
#define NUM_WRITERS 2

pthread_rwlock_t rwlock;
int shared_data = 0;

void* reader(void* arg) {
    int id = *((int*)arg);
    while (1) {
        pthread_rwlock_rdlock(&rwlock);
        printf("Reader %d: read sharedData = %d\n", id, shared_data);
        pthread_rwlock_unlock(&rwlock); 
        sleep(1);
    }
    return NULL;
}

void* writer(void* arg) {
    int id = *((int*)arg);
    while (1) {
        pthread_rwlock_wrlock(&rwlock);
        shared_data++;                    
        printf("Writer %d: wrote sharedData = %d\n", id, shared_data);
        pthread_rwlock_unlock(&rwlock);
        sleep(2);
    }
    return NULL;
}

int main() {
    pthread_t readers[NUM_READERS], writers[NUM_WRITERS];
    int reader_ids[NUM_READERS], writer_ids[NUM_WRITERS];

    pthread_rwlock_init(&rwlock, NULL);

    for (int i = 0; i < NUM_READERS; i++) {
        reader_ids[i] = i + 1;
        pthread_create(&readers[i], NULL, reader, &reader_ids[i]);
    }

    for (int i = 0; i < NUM_WRITERS; i++) {
        writer_ids[i] = i + 1;
        pthread_create(&writers[i], NULL, writer, &writer_ids[i]);
    }

    for (int i = 0; i < NUM_READERS; i++) {
        pthread_join(readers[i], NULL);
    }
    for (int i = 0; i < NUM_WRITERS; i++) {
        pthread_join(writers[i], NULL);
    }

    // 销毁读写锁
    pthread_rwlock_destroy(&rwlock);

    return 0;
}

我实现了一个队列，多个线程往里面放数据、取数据，用互斥还是读写呢？ 都不好，C 里的链表操作都是指针，挂个指针耗不了多少时间，稍稍等一下就行了。自旋锁就是干这个的。

自旋锁

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

#define NUM_THREADS 4

pthread_spinlock_t spinlock;
int sharedCounter = 0;

void* incrementCounter(void* arg) {
    for (int i = 0; i < 100000; i++) {
        pthread_spin_lock(&spinlock);
        sharedCounter++;
        pthread_spin_unlock(&spinlock);
    }
    return NULL;
}

int main() {
    pthread_t threads[NUM_THREADS];

    pthread_spin_init(&spinlock, PTHREAD_PROCESS_PRIVATE);

    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_create(&threads[i], NULL, incrementCounter, NULL);
    }

    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    printf("Final counter value: %d\n", sharedCounter);

    pthread_spin_destroy(&spinlock);

    return 0;
}

可能是每个线程都在频繁地获取和释放锁。如果锁的争用很高，自旋锁的优势会被削弱，在自旋期间消耗大量 CPU 时间。看来在多核上，自旋锁会在同一 CPU 核心上运行，减少了缓存一致性问题有优势了。

自旋锁的特点，可以看出，防止挂起导致的上下文切换，如果锁的持有时间很短或竞争不激烈，那么非常适合用自旋锁，比如队列节点的添加和删除上。

条件变量

前面在互斥那里遇到过，我可以保护这个变量不会被多个线程同时访问，但我阻止不了线程调度啊。就是说我决定不了执行顺序，那么条件变量就用到了。

#include <stdio.h>
#include <pthread.h>
#include <unistd.h>
#include <signal.h>
#include <stdbool.h>

volatile sig_atomic_t stop = false;
volatile sig_atomic_t running = true;  /* to check at every loop */

pthread_mutex_t mutex;
pthread_cond_t not_empty;
pthread_cond_t not_full;

#define PRODUCT_SIZE 5
int goods[PRODUCT_SIZE];
int count = 0;

void* producer(void *)
{
    while (running) {
        pthread_mutex_lock(&mutex);

        while (PRODUCT_SIZE == count) {
            pthread_cond_wait(&not_full, &mutex);
        }

        goods[count++] = 1;
        printf("[P]:  %d\n", count);

        pthread_cond_signal(&not_empty);
        pthread_mutex_unlock(&mutex);

        sleep(2);
    }

    printf("producer exit...\n");
    return NULL;
}

void* consumer(void *)
{
    while (running) {
        pthread_mutex_lock(&mutex);

        while (0 == count) {
            pthread_cond_wait(&not_empty, &mutex);
        }

        goods[count--] = 0;
        printf("[C]: %d\n", count);

        pthread_cond_signal(&not_full);
        pthread_mutex_unlock(&mutex);

        sleep(1);
    }

    printf("consumer exit...\n");
    return NULL;
}

void signal_handle(int sig)
{
    stop = 1;
}

int main()
{
    pthread_t t1, t2;
    pthread_mutex_init(&mutex, NULL);
    pthread_cond_init(&not_empty, NULL);
    pthread_cond_init(&not_full, NULL);
    signal(SIGINT, signal_handle);

    pthread_create(&t1, NULL, producer, NULL);
    pthread_create(&t2, NULL, consumer, NULL);


    while (!stop) {
        sleep(1);
    }

    running = false;

    pthread_join(t1, NULL);
    pthread_join(t2, NULL);

    pthread_mutex_destroy(&mutex);
    pthread_cond_destroy(&not_empty);
    pthread_cond_destroy(&not_full);
    return 0;
}

我想要所有线程都准备好了（主线程给他们配置好），然后一起运行，怎么弄呢？设置一道起跑线吧，叫做屏障（barrier）。

屏障

#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>
#include <unistd.h>

#define NUM_THREADS 4

pthread_barrier_t barrier;
int shared_data = 0;

void* threadFunction(void* arg) {
    int id = *((int*)arg);

    printf("Thread %d is doing work...\n", id);
    sleep(1);

    pthread_barrier_wait(&barrier);

    printf("Thread %d has passed the barrier.\n", id);
    shared_data += id;

    return NULL;
}

int main() {
    pthread_t threads[NUM_THREADS];
    int thread_ids[NUM_THREADS];

    pthread_barrier_init(&barrier, NULL, NUM_THREADS);

    for (int i = 0; i < NUM_THREADS; i++) {
        thread_ids[i] = i + 1;
        pthread_create(&threads[i], NULL, threadFunction, &thread_ids[i]);
    }

    for (int i = 0; i < NUM_THREADS; i++) {
        pthread_join(threads[i], NULL);
    }

    printf("Final shared data value: %d\n", shared_data);

    pthread_barrier_destroy(&barrier);

    return 0;
}

Finally

最后，来看看，

  • 互斥，只有两种状态，锁住、解锁。
  • 读写，三状态。读锁、写锁、解锁。写时互斥（其余写锁、读锁被阻塞），读时共享（写锁一定被阻塞，其余读锁不会）。
  • 自旋锁，互斥加锁失败，线程会休眠挂起切换，自旋加锁不会休眠挂起切换，自旋，就是等一等的意思:)。
  • 条件变量，不竞争，等待条件的发生。但需要互斥去保护条件变量，条件好了发出信号。
  • 屏障，一组线程到达某个点，集合齐了后，再继续执行。其实，pthread_join 就是一种特殊的屏障，如果你把 join 放在一起的话。

最后，还有一种讨厌的情况，死锁（deadlock）。

对一把锁连续加锁两次会死锁。

两个线程1、2，两把锁 A、B，线程 1 中，先锁 A 再锁 B,线程 2 中先锁 B 再锁 A，就会产生尴尬的死锁场面。表面上，是锁的调用顺序不一样，实际上，是两个线程都在请求另一个线程锁住的资源。
